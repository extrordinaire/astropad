        \begin{array}{r}
          \bin{   1000 0000} = A \\
        + \bin{   1101 0000} = B \\
          \hline
          \bin{\ovf{1} 0101 0000} = C
        \end{array}
    $$  
    
    C = `1` `0101_0000`$_2$ $\implies$ Overflow = `1`$_2$

    $$ 
    \begin{align*}
      \overline{B} &= &\bin{0010 1111} \\
                   &+ &\bin{1} \\
                   \hline
      \cii{B}     &= &\bin{0011 0000}
    \end{align*}
    $$

    $$
        \begin{align*}
          \bin{  1000 0000} &= A \\
        + \bin{  0011 0000} &= \cii{B} \\
          \hline
          \bin{  1011 0000} &= \cii{C}
        \end{align*}
    $$

    $$
      \begin {align*}
        \cii { C}            &= \ci{C} &+ \bin { 1} \\
        \bin { 1011 0000} &= \ci{C} &+ \bin { 1} \\
        -\bin { 1}        &= &- \bin { 1} \\
        \hline
        \bin { 1010 1111} &= \ci{C} \\
        \bin { 0101 0000} &= C

      \end{align*}
    $$

  </li>
